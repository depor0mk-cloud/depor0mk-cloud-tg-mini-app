        </div>

        <button onclick="buyFactory()" id="factory-card" class="w-full bg-white/5 border border-white/5 p-4 rounded-3xl flex items-center justify-between active-scale transition-all">
            <div class="flex items-center gap-4 text-left">
                <div class="p-3 bg-blue-500/10 rounded-2xl">
                    <i data-lucide="factory" class="text-blue-400 w-6 h-6"></i>
                </div>
                <div>
                    <p class="font-bold text-sm">HAKOS ЗАВОД</p>
                    <p class="text-[10px] text-neutral-500" id="fact-lvl">Уровень: 0</p>
                </div>
            </div>
            <div class="text-right">
                <p class="text-green-400 font-black" id="fact-price">50</p>
                <p class="text-[8px] text-neutral-600 font-bold uppercase">Очков</p>
            </div>
        </button>
    </div>

    <script>
        // Интеграция с Telegram
        const tg = window.Telegram.WebApp;
        tg.expand();
        tg.headerColor = '#000000';

        // Данные
        let score = parseInt(localStorage.getItem('hakos_s')) || 0;
        let factories = parseInt(localStorage.getItem('hakos_f')) || 0;
        const startPrice = 50;
        const growth = 1.7;

        // UI Элементы
        const scoreDisplay = document.getElementById('score');
        const incomeDisplay = document.getElementById('income-rate');
        const fPriceDisplay = document.getElementById('fact-price');
        const fLevelDisplay = document.getElementById('fact-lvl');
        const mainBtn = document.getElementById('main-btn');

        function refresh() {
            scoreDisplay.innerText = Math.floor(score).toLocaleString();
            incomeDisplay.innerText = `ДОХОД: +${factories * 2}/СЕК`;
            fLevelDisplay.innerText = `Уровень: ${factories}`;
            
            const nextP = Math.floor(startPrice * Math.pow(growth, factories));
            fPriceDisplay.innerText = nextP.toLocaleString();

            // Сохранение в память браузера
            localStorage.setItem('hakos_s', score);
            localStorage.setItem('hakos_f', factories);

            // Визуальный эффект доступности покупки
            const card = document.getElementById('factory-card');
            if (score < nextP) {
                card.style.opacity = "0.4";
            } else {
                card.style.opacity = "1";
            }
        }

        // Клик
        mainBtn.addEventListener('click', (e) => {
            score += 1;
            tg.HapticFeedback.impactOccurred('light'); // Вибрация
            showPlusOne(e.clientX, e.clientY);
            refresh();
        });

        function showPlusOne(x, y) {
            const div = document.createElement('div');
            div.className = 'floating-pts';
            div.innerText = '+1';
            div.style.left = x + 'px';
            div.style.top = y + 'px';
            document.body.appendChild(div);
            setTimeout(() => div.remove(), 700);
        }

        // Покупка
        window.buyFactory = () => {
            const nextP = Math.floor(startPrice * Math.pow(growth, factories));
            if (score >= nextP) {
                score -= nextP;
                factories++;
                tg.HapticFeedback.notificationOccurred('success');
                refresh();
            } else {
                tg.HapticFeedback.notificationOccurred('error');
            }
        }

        // Автокликер (1 раз в секунду)
        setInterval(() => {
            if (factories > 0) {
                score += (factories * 2);
                refresh();
            }
        }, 1000);

        lucide.createIcons();
        refresh();
    </script>
